UC2:
create  table  employee_payroll
(
id        INT Unsigned NOT NULL AUTO-INCREMENT,
name     VARCHAR(150) NOT NULL,
salary     Double  NOT NULL,
start        DATE  NOT   NULL,
PRIMARY  KEY    (id)
);
